How to BEJY

  Webapps config

This show the simplest config to run only webapps. 

<!--    BEJY V1.1.1.34 config file for using a webapps directory     -->
<bejy>
  <global
    logFile="logs/default"
    javac="c:\jdk1.3\bin\javac.exe -g"
  />
  <!-- Config for the http server - each server may have a config section in this file -->
  <http>
    <host name="*" logFile="logs/localhost">
      <webapps path="webapps" />
    </host>
  </http>
<!-- config of a http server -->
  <server
    logFile="logs/http"
    name="http"
    port="80"
    protocol="de.bb.bejy.http.HttpFactory"
  >
    <threads max="999" wait="1" />
  </server>
  <!-- and some mime types - global for all servers -->
  <mime-types>
    <mime-type name="gif"  value="image/gif" />
    <mime-type name="html" value="text/html" />
    <mime-type name="js"   value="text/javascript" />
  </mime-types>
</bejy>
